**date locale helper:**
```ts
// frontend/src/lib/date-locale.ts
import { enUS, zhCN } from 'date-fns/locale';
import type { Locale } from 'date-fns';
import { useLocale as useIntlLocale } from 'next-intl';

const dateFnsLocales: Record<string, Locale> = {
  'en-US': enUS,
  'zh-CN': zhCN,
};

export function getDateFnsLocale(intlLocale: string): Locale {
  return dateFnsLocales[intlLocale] ?? enUS;
}

// for non-React modules
export const useDateFnsLocale = () => getDateFnsLocale(useIntlLocale());
```
